package com.company;

public class StringOperations {
    public static void main(String[] args) {
        String word = "orGANization";

        //uppercase and lowercase

        System.out.println(word.toUpperCase());
        System.out.println(word.toLowerCase());

        //0 1 2 3 4 5
        //l e n g t h
        // -6 -5 -4 -3 -2 -1

        //length
        System.out.println(word.length());


        //index of a character
        System.out.println(word.indexOf("N"));



        //character at a particular index
        System.out.println(word.charAt(8));


        //slice out index
        word.substring(0,7);
        System.out.println(word.substring(0,7));


        //word from the back
        word.charAt(word.length()-1);
        System.out.println(word.charAt(word.length()-1));


        //democracy tasks
        String word2 = "DEMOCRACY";
        String newword = "EMOC";


        //d e m o c r a c y
        //0 1 2 3 4 5 6 7 8
        // d  e  m  o  c  r  a  c  y
        //-9 -8 -7 -6 -5 -4 -3 -2 -1

        //slice out index COME
        word2.substring(1,5);

        String rev = new StringBuilder(newword).reverse().toString();
        System.out.println("Reverse String is : "+ rev);







    }
}
