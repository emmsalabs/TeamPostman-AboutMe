package com.company;

public class JavaMethods {
    public static void main(String[] args) {
        //to access the properties of a class we create an object of the class
        JavaMethods methodVar = new JavaMethods();
        methodVar.printName("Gbenga");
        methodVar.myAge(100);
        int userAge = methodVar.ageCalculation(1997, 2022);
        System.out.println(userAge+ " is the calculated age.");



    }
    public void printName (String name){
        System.out.println("My name is "+ name);
    }
    public void myAge (int age){
        System.out.println("My age is "+ age);
    }
    public int ageCalculation(int dob, int presentYear){
        int age = presentYear - dob;
        return age;


    }




}
