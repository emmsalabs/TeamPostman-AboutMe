package com.company;

public class BasicArithmeticOps {
    public static void main(String[] args) {
        //basic arithmetic
        byte age1 = 20;
        byte age2 = 30;

        double age3 = 20;
        double age4 = 30;

        System.out.println(age1+age2);

        System.out.println(age1-age2);

        System.out.println(age3/age4);

        System.out.println(age1*age2);


        age1++;
        // incremental ++
        System.out.println(age1);

        age2--;
        //decremental --
        System.out.println(age2);

        age3++;
        //incremental ++
        System.out.println(age3);

        age4--;
        //decremental --
        System.out.println(age4);



        //compound assignment operators -= += /= *=
        age1 -= age2;
        //20 - 30
        //age1 = -10
        System.out.println(age1);


        age1 += age2;
        //20 + 30
        //age1 = 50
        System.out.println(age1);


        age3 /= age4;
        //20 / 30
        //age3 = 0.72
        System.out.println(age3);


        age3 *= age4;
        //20 * 30
        //age3 = 600
        System.out.println(age3);



        //Area of a circle is pie r2
        double radius1 = 12.7;
        double radius2 = 12.7;
        double pie = 3.142;
        String unit = "ms";

        System.out.println(pie*radius1*radius2 + unit);

    }
}
